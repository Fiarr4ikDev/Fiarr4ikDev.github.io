SQL Заметки

Создание резервной копии 


BACKUP DATABASE [YourDatabaseName]
TO DISK = 'C:\Backup\YourDatabaseName.bak'
GO

Восстановление из копии 


RESTORE DATABASE [YourDatabaseName]
FROM DISK = 'C:\Backup\YourDatabaseName.bak'
GO

C# Заметки

Подключение к базе данных 


string connectionString = @"Server=ваш_сервер;Database=название_бд;User Id=sa;Password=пароль_при_создании_сервера;";

Выполнение SQL-запроса 


string query = @"тут запрос";
SqlConnection connection = new SqlConnection(connectionString);  // connectionString - строка подключения к бд
SqlDataAdapter adapter = new SqlDataAdapter(query, connection); // query - запрос

DataTable table = new DataTable();
adapter.Fill(table);
dataGridViewData.DataSource = table; //dataGridViewData - имя DataGridView
